114inline static String fakeAPLib::getCurrentTime()
115{
116 unsigned long runMillis= millis();
117 unsigned long allSeconds= runMillis / 1000;
118 int secsRemaining= allSeconds % 3600;
119
120 uint8_t hours = allSeconds / 3600;
121 uint8_t minutes = secsRemaining / 60;
122 uint8_t seconds = secsRemaining % 60;
123
124 return String(hours) + ":" + String(minutes) + ":" + String(seconds);
125}
